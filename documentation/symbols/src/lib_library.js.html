171     /** Perspective projection, imitates gluPerspective() */
172     function perspective(fov, ar, near, far) {
173         var f = 1/Math.tan(fov/2);
174         var div = near - far;
175         return [f/ar, 0,  0,               0,
176                 0,    f,  0,               0,
177                 0,    0,  (far+near)/div, -1,
178                 0,    0,  2*far*near/div,  0 ];
179     };

256     /** 4x4 Matrix multiplication */
257     function matmul(a,b){
258         var i,j,k,m = zeros();
259         for (i=0;i<4;i++){
260             for (j=0;j<4;j++){
261                 for(k=0;k<4;k++){
262                     m[j*4+i] += a[k*4+i]*b[j*4+k];
263                 }
264             }
265         }
266         return m;
267     }
268 
269     /** 4x4 Matrix times 4x1 vector multiplication */
270     function matvec(a,b){
271         var i,j,res = [0,0,0,0];
272         for (i=0;i<4;i++){
273             for(j=0;j<4;j++){
274                 res[i] += a[j*4+i]*b[j];
275             }
276         }
277         return res;
278     }

280     /**
281      * 4x4 matrix times 4xN matrix multiplication. Does a bit of extra
282      * work but allows the same routine to multiply both matrices and
283      * vectors. The end result seems to be 15 bytes shorter when using
284      * this instead of separate matmul and matvec routines.
285      */
286     function matmul4(a,b){
287         var i,j,k,m=[];
288         for (i=0;i<b.length;i++){
289             m[i]=0;
290         }
291         for (i=0;i<4;i++){
292             for (j=0;j<(b.length/4);j++){
293                 for(k=0;k<4;k++){
294                     m[j*4+i] += a[k*4+i]*b[j*4+k];
295                 }
296             }
297         }
298         return m;
299     }

310     /**
311      * Transpose the upper 3x3 part and zero the rest. For building
312      * the normal matrix from the inverse of model matrix.

314     function transposed3x3(a){
315         return [a[0],a[4],a[8], 0,
316                 a[1],a[5],a[9], 0,
317                 a[2],a[6],a[10],0,
318                 0,   0,   0,   1];

321     /** Cross product for homogeneous directions. "[(axb)^t,0]^t" */
322     function cross(a,b){
323         return [a[1]*b[2]-a[2]*b[1],
324                 a[2]*b[0]-a[0]*b[2],
325                 a[0]*b[1]-a[1]*b[0],
326                 0];
327     }

329     /** Normalize x,y,z disregarding and untouching w */
330     function nmld(v){
331         var length3d=Math.sqrt(v[0]*v[0]+v[1]*v[1]+v[2]*v[2]);
332         return [v[0]/length3d,v[1]/length3d,v[2]/length3d,v[3]];
333     }

335     /** Homogenize by x/w,y/w,z/w  */
336     function homogenized(v){
337         if (v[3]==0) return v;
338         return [v[0]/v[3],v[1]/v[3],v[2]/v[3],1];
339     }

370 // Traversal without inverse matrices and thus normal matrices. Use
371 // this if you know you don't need normals or camera as scenegraph node.
372 function traverse(node,ms){
373     ms=node.f.reduce(matmul4,ms);
374     gl.uniformMatrix4fv(
375         gl.getUniformLocation(prg,"mv"), false, ms);
376     node.o.map(function(o){o.c(gl);}); // map < forEach :)
377     node.c.map(function(c){traverse(c,ms);});
378 }

380 /**
381  * Traverse a scene graph, and output to the WebGL pipeline.
382  *
383  * Algorithm:
384  *
385  *  This is a very basic recursive tree traversal.
386  *
387  * Size optimizations:
388  *
389  *   - The 'functional' operations provided by Javascript allow a very
390  *     compact way of writing this.
391  *
392  *   - Call map() without using its output. It is wasteful, but it is
393  *     4 characters shorter to write than forEach().
394  *
395  *   - Always send two uniform matrices, regardless of whether
396  *     anything is drawn using them. Wasted effort when not drawing.
397  **/
398 function traverse_wi(node,ms){
399     ms=node.f.reduce(matmul_wi,ms);
400     gl.uniformMatrix4fv(
401         gl.getUniformLocation(prg,"mv"), false, ms);
402     gl.uniformMatrix4fv(
403         gl.getUniformLocation(prg,"nm"), false, transposed3x3(ms.n));
404     node.o.map(function(o){o.c(gl);}); // map < forEach :)
405     node.c.map(function(c){traverse_wi(c,ms);});
406 }

429     /**
430      * Create an evaluator that can return a local frame for circle in
431      * the xy plane. As in the MIT course assignment, but transposed.
432      */
433     function funCircleBasic(radius,n){
434         var r = radius;
435         this.n = n; // Fidelity hint for the surface evaluator
436         this.c = function(t){
437             var s = Math.sin(t * 2 * PI);
438             var c = Math.cos(t * 2 * PI);
439             return [-c,  -s,   0,  0,  // normal
440                      0,   0,   1,  0,  // binormal
441                     -s,   c,   0,  0,  // tangent
442                      c*r, s*r, 0,  1   // position
443                    ];
444         };
445     }

447     /**
448      * Create an evaluator that can return a local frame for a part or
449      * whole circle on the xy plane. The starting point of the arc is
450      * at the highest y point "top", and direction is counterclockwise
451      * in right-handed coordinates, z towards viewer. Assumes
452      * 0<arclen<=1. Arclen is optional.
453      */
454     function funCircle(radius,swpfidel,arclen){
455         var r = radius;
456         var a = arclen?arclen:1;
457         // Fidelity hint for the sweep surface evaluator
458         this.n = swpfidel?swpfidel:10;
459         this.c = function(t){
460             var s = Math.sin(t * 2 * PI * a);
461             var c = Math.cos(t * 2 * PI * a);
462             return [    s,  -c,  0,  0,  // normal
463                         0,   0,  1,  0,  // binormal
464                        -c,  -s,  0,  0,  // tangent
465                      -s*r, c*r,  0,  1   // position
466                    ];
467         };
468     }

486     /**
487      * Simple uniform B-spline evaluator.
488      *
489      * Hmm.. position and tangent could be evaluated for any t... But
490      * how to maintain correct normal and bi-normal for surface
491      * creation? Initial idea: pre-compute at some intervals using the
492      * cross product trick from our lecture notes, and then evaluate a
493      * normalized interpolant upon call to eval(). NOTE: Only need to
494      * store binormal (?), because normal can be computed via cross
495      * product. The binormal rotations could be corrected while
496      * precomputing. (not yet done).
497      *
498      * Note that analytic tangent doesn't exist for all possible
499      * inputs - we don't handle curves with vanishing derivatives,
500      * so keep this in mind when defining control points.
501      *
502      * TODO: Maybe could be sloppy and not even interpolate, if result
503      * is OK visually? Hmm... we do have storage, so why not just
504      * precompute like a *lot* of values when creating the spline
505      * object, and then return the closest one in compute(t)? Nasty, but
506      * without redundance...
507      *
508      * Back-of-the-envelope: 100 control pts * 100 intermediates * 16
509      * * 8 byte float is 1280000 bytes... well.. that's a megabyte for
510      * one spline..  admittedly, it sounds like a lot..
511      *
512      * TODO (cheating a bit, though): Use only xy-curves with no
513      * change of curvature, and delete all the code about flipping
514      * gradients.  Would be so much smaller and leaner! Well.. if we
515      * don't cheat that much, then at least the production should USE
516      * the feature and have some twisty curve(s) to show it
517      * off.. Closure will omit unused functions, so we can have many
518      * versions here in the library. This is the bloaty, most general
519      * version. Depending on what you need, you can "cheat" more by
520      * using the restricted versions below.
521      */

523     function funBSpline(pts) {
524         // Let us declare all vars here to shorten the code.
525         var i;
526         var g,Tt,dTt,v,T,N;
527         var t,ifirst,npts = this.n = pts.length/4;
528         var nsteps = (npts-3)*300; // internal points (actually +1 because we go to t=1.0)
529         var b = [];           // the internal storage
530         var B = [0,0,1,0];    // "arbitrary binormal at beginning"
531         //this.n=npts;
532 
533         for (i=0;i<nsteps+1;i++){
534             t = i/nsteps; // scale t to [0,1] within curve
535             if (i<nsteps) {
536                 ifirst = t*(npts-3) | 0; // 1st point is... (funny "|0" makes a floor())
537                 t = t*(npts-3) - ifirst; // reset t to [0,1] within segm.
538             } else {
539                 ifirst = npts-4;
540                 t = 1;
541             }
542 
543             g = pts.slice(ifirst*4,ifirst*4+16); // pick cps to G.
544             Tt = [1, t, t*t, t*t*t];
545             dTt = [0, 1, 2*t, 3*t*t];
546             v = matmul4(matmul4(g, bspB),Tt);
547             T = nmld(matmul4(matmul4(g, bspB),dTt));
548             N = nmld(cross(B,T));
549             B = nmld(cross(T,N));
550             b.push([].concat(N, // "normal"/orientation
551                              B, // binormal
552                              T, // tangent
553                              v)); // pos.
554         }
555 
556         // as of now, we don't care to interpolate:
557         this.c = function(t){
558             return b[0 | t*(nsteps)];
559         }
560     }

650     function funBezierCurve(pts) {
651         // Let us declare all vars here to shorten the code.
652         var i;
653         var g,Tt,dTt,v,T,N;
654         var t,ifirst,npts = pts.length/4;
655         this.n = pts.length*4; // Fidelity hint for surface eval
656         var nsteps = ((npts-1)/3)*10; // internal points (actually +1 because we go to t=1.0)
657         var b = [];           // the internal storage
658         var B = [0,0,1,0];    // "arbitrary binormal at beginning"
659         //this.n=npts;
660 
661         for (i=0;i<nsteps+1;i++){
662             t = i/nsteps; // scale t to [0,1] within curve
663             if (i<nsteps) {
664                 ifirst = (t*((npts-1)/3) | 0)*3; // 1st point is... (funny "|0" makes a floor())
665                 t = t*((npts-1)/3) - (ifirst/3); // reset t to [0,1] within segm.
666             } else {
667                 ifirst = npts-4;
668                 t = 1;
669             }
670 
671             g = pts.slice(ifirst*4,ifirst*4+16); // pick cps to G.
672             Tt = [1, t, t*t, t*t*t];
673             dTt = [0, 1, 2*t, 3*t*t];
674             //v = matmul4(matmul4(g, bezB),Tt);
675             v = homogenized(matmul4(matmul4(g, bezB),Tt));
676             T = nmld(matmul4(matmul4(g, bezB),dTt));
677             N = nmld(cross(B,T));
678             B = nmld(cross(T,N));
679             b.push([].concat(N, // "normal"/orientation
680                              B, // binormal
681                              T, // tangent
682                              v)); // pos.
683             //console.log("ja tuota " + npts + " " + ifirst + " " + t);
684         }
685 
686         // as of now, we don't care to interpolate:
687         this.c = function(t){
688             return b[0 | t*(nsteps)];
689         }
690     }
