 11 var SAMPLE_RATE = 44100;
 12 var startTimeInMillis = null;
 13 
 14 var dbg_ms_at_last_seek = null; //DEBUG
 15 var dbg_ms_offset = 0;          //DEBUG
 16 var dbg_paused = false;         //DEBUG

 18 /**
 19  * This is the event loop that happens on every frame.
 20  *
 21  * Assumes prior initialization of C (canvas).
 22  *
 23  * Assumes prior initialization of gl and prg.
 24  *
 25  * Debug mode needs initialization of dbg_xxx
 26  *
 27  */
 28 var loopfunc = function(curTimeInMillis)
 29 {
 30     try                                                          //DEBUG
 31     {                                                            //DEBUG
 32         // Time of this frame; fix beginning upon first entry.
 33  	    if (!startTimeInMillis) startTimeInMillis = curTimeInMillis;
 34         // In compo mode, compute t as the music beat since start of show:
 35         var t = (curTimeInMillis - startTimeInMillis) * songBeatsPerMinute / 60000;
 36 
 37         // In debug mode, time is a bit more elaborate, because we want to
 38         // be able to seek back and forth.
 39         if (!dbg_ms_at_last_seek) dbg_ms_at_last_seek = startTimeInMillis; //DEBUG
 40         var dbg_ms_since_last_seek = curTimeInMillis - dbg_ms_at_last_seek; //DEBUG
 41         if (dbg_paused) dbg_ms_since_last_seek = 0; //DEBUG
 42         t = (dbg_ms_offset + dbg_ms_since_last_seek) * songBeatsPerMinute / 60000; //DEBUG
 43 
 44         // Update canvas size (window object is implicit; need no "window.X")
 45         var w = innerWidth, h = innerHeight;
 46         if (w != Cw || h != Ch) {
 47             gl.viewport(0, 0, Cw=C.width=w, Ch=C.height=h);
 48         }
 49 
 50         /* In debug mode, show window size, aspect ratio, and time.       */
 51         /* Omit info if the URL ends in '#'. Use for tidy screenshots...  */
 52         dbg_url=window.location.href;                                //DEBUG
 53         if (dbg_url.substring(dbg_url.length -1)!='#'){              //DEBUG
 54             dbg_show_aspect.nodeValue="Size: "+w+"x"+h+" "+w/h;      //DEBUG
 55             dbg_show_time.nodeValue=" time=" + Math.floor(audio_time / SAMPLE_RATE) +"s" /*(audio.currentTime|0)*/ //DEBUG
 56                                          + "(beat " +(t|0)+ ")";     //DEBUG
 57         }                                                            //DEBUG
 58 
 59         // Update the HTML and CSS parts, if you wanna have something like text..
 60         updateDocument(t);
 61 
 62         // Re-build the scenegraph for every frame (can animate):
 63         var sceneroot = buildSceneAtTime(t);
 64 
 65         // Scene is built. Then we actually draw it.
 66         // Clear buffer
 67         gl.clearColor(clearColor[0], clearColor[1], clearColor[2], clearColor[3]);
 68         gl.clear(gl.DEPTH_BUFFER_BIT | gl.COLOR_BUFFER_BIT);
 69 
 70         // Transfer perspective matrix to shader:
 71         var persmat = perspectiveFhc(5,w/h);
 72         gl.uniformMatrix4fv(gl.getUniformLocation(prg,"p"), false, persmat);
 73 
 74         // TODO: Could switch the shader based on time / object properties:
 75         gl.useProgram(prg);
 76 
 77         // Then we display the scenegraph
 78         findcam_wi(sceneroot,rotX_wi(0));
 79 
 80         // Set light position (very naive as of yet):
 81         // TODO: Multiple lights, included in the scenegraph as "Light objects".
 82         // As of now, we have to hardcode light position separately from the scene transforms:
 83 	/* FIXME: 
 84         gl.uniform4fv(gl.getUniformLocation(prg, "l"),
 85                       matmul4(cameraTransformation,[1,.1,1.5,1]));
 86         */
 87         gl.uniform4fv(gl.getUniformLocation(prg, "l"),
 88 		      defaultLightDirection);
 89 
 90         traverse_wi(sceneroot,cameraTransformation);
 91 	    requestAnimationFrame(loopfunc);
 92 
 93     }                                                    //DEBUG
 94     catch (err)                                          //DEBUG
 95     {                                                    //DEBUG
 96         alert("Error: " + err.message);                  //DEBUG
 97     }                                                    //DEBUG
 98 };

128     document.body.appendChild(dbgInfoDiv);           //DEBUG

138     // Apply p01's trick for grabbing short names from GL obj
139     // (http://slides.com/pdesch/js-demoscene-techniques#/5/6)
140     // This didn't help me earlier when trying to make a 1k..
141     // is OK now for 4k. saves something like 30 bytes / 4kb.
142     // Not much.. The trick itself costs some 45 bytes compressed.
143     // In 2023 I'm safeguarding against name clashes, so the regex
144     // changes a bit from earlier versions.
145     var dbgNameLists = {}; //DEBUG
146     for(s in gl){
147         // Instanssi 2023 still had: gl[s.match(/^..|[A-Z]|\d\D+$/g).join('')]=gl[s];
148         // Contemplation at some point.. some alternatives with few clashes:
149         // gl[s.match(/^.|[A-Zhol]|\d|[ruiv]+$/g).join('')]=gl[s];
150         //gl[s.match(/^.|[A-Zlp]|\d.|[fv]+$/g).join('')]=gl[s];
151         //gl[s.match(/^.|[A-Zlp]|\d.*$/g).join('')]=gl[s];
152         //gl[s.match(/^.|[A-Zlp\d]|[fuiv]+$/g).join('')]=gl[s];
153 
154         // The trick itself:
155         const trick_regex = /^.|[A-Zlp1-4]|[fuiv]*$/g;
156         gl[s.match(trick_regex).join('')] = gl[s];
157         // Gather names for clash inspection and creating a minifier:    //DEBUG
158         var shortname = s.match(trick_regex).join('');                   //DEBUG
159         if (shortname in dbgNameLists) {dbgNameLists[shortname].push(s); //DEBUG
160         } else { dbgNameLists[shortname]=[s]; }                          //DEBUG
161     }
162     // Inspect all names and clashing ones //DEBUG
163     var dbgClashing = {};               //DEBUG
164     for(s in dbgNameLists)              //DEBUG
165         if ((dbgNameLists[s].length>1)  //DEBUG 
166             && (!dbgNameLists[s][0].match(/^[A-Z]/g))) //DEBUG
167                 dbgClashing[s]=dbgNameLists[s]; //DEBUG
168     console.log(dbgNameLists);          //DEBUG
169     console.log(dbgClashing);           //DEBUG
170     // Output a sed script to change original names that don't clash:   //DEBUG
171     // Then we should be safe... Can use any name; minified if possible //DEBUG
172     var dbgSedStrings=[]; //DEBUG
173     for (s in dbgNameLists) {           //DEBUG
174         if ((dbgNameLists[s].length==1) //DEBUG
175             && (!dbgNameLists[s][0].match(/^[A-Z]/g)) //DEBUG
176             ) {dbgSedStrings.push("s/gl\\."+dbgNameLists[s][0]+"(/gl."+s+"(/g"); //DEBUG
177         }                               //DEBUG
178     }                                   //DEBUG
179     // console.log(dbgSedStrings.join('\n')); //DEBUG

187     // TODO: Multiple, switchable shaders!

191     gl.shaderSource(s, vertex_shader_minimal_with_normalmatrix);
192     //gl.shaderSource(s, test_vert);
193     gl.compileShader(s);
194     if (!gl.getShaderParameter(s, gl.COMPILE_STATUS))                 //DEBUG
195         alert("Vertex shader: "+ gl.getShaderInfoLog(s));             //DEBUG
196     gl.attachShader(prg = gl.createProgram(), s);
197 
198     s = gl.createShader(gl.FRAGMENT_SHADER);
199     gl.shaderSource(s, fragment_shader_camspace_directed_with_white_fog);
200     //gl.shaderSource(s, fragment_shader_camspace_directed);
201     //gl.shaderSource(s, fragment_shader_pointlight_cameraspace);
202     //gl.shaderSource(s, noisz_frag);
203     gl.compileShader(s);
204     if (!gl.getShaderParameter(s, gl.COMPILE_STATUS))                 //DEBUG
205         alert("Fragment shader: "+ gl.getShaderInfoLog(s));           //DEBUG
206     gl.attachShader(prg, s);
207 
208     gl.linkProgram(prg);
209     if (!gl.getProgramParameter(prg, gl.LINK_STATUS))                 //DEBUG
210         alert("Link program: "+ gl.getProgramInfoLog(prg));           //DEBUG
211 
212     // Initialization code is now separated production-wise. Easier to find for editing.
213     initAssets();
214     // Things like scolltext could initialized in this call, also on product side:
215     initDocument();
216 
217 
218     /* Initialize song. */
219     var player = new CPlayer();
220     player.init(song);
221     while (player.generate() < 1){};
222 
223     var audio_time=0;
224 
225     /**
226      * The onaudioprocess handler that gets called for audio output. 
227      * This will add to audio_time which is much coarser than animation frame
228      * rate.
229      * 
230      * Debug mode enables seeking and pausing with clicks by means of updating
231      * audio_time and aligning actual animation time to it.
232      * Debug mode outputs silence here when paused.
233      */
234     function audioHandler(event){
235         if (dbg_paused) event.outputBuffer.getChannelData(0).fill(0); else { //DEBUG
236          
237         player.cpy(audio_time, 4096, event.outputBuffer.getChannelData(0));
238         audio_time += 4096;
239 
240         } //DEBUG
241     };
242 
243 
244     /* Nowadays, browsers allow media only upon user event, so click pls: */
245     // I used to have _document.onclick here, but Timo taught me that the
246     // window itself has an onclick property, so let's use that one...
247     onclick=()=>
248         {
249             // Accidental double clicks make it a mess; prevent that:
250             onclick = null;
251         /* In debug mode I want to control the fullscreen myself, so iffalse..*/
252         if (false)                                     //DEBUG
253             C.style.cursor='none';
254         if (false)                                     //DEBUG
255             C.requestFullscreen();
256 
257  	    var audioctx = new AudioContext({sampleRate:SAMPLE_RATE});
258  	    var sp = audioctx.createScriptProcessor(4096, 0, 1);
259  	    sp.connect(audioctx.destination);
260  	    sp.onaudioprocess = audioHandler;
261  	    requestAnimationFrame(loopfunc);
262 
263         /* In debug mode I want to be able to seek, so wire a callback : */
264         C.addEventListener("click", function(e){             //DEBUG
265             // Audio seek by just setting sample index:      //DEBUG
266             audio_time =                                     //DEBUG
267                 e.pageX/C.width*player.ns() | 0;             //DEBUG
268             // Handle seek and pausing in debug mode:
269             dbg_ms_offset = audio_time / SAMPLE_RATE * 1000; //DEBUG
270             dbg_ms_at_last_seek = performance.now();         //DEBUG
271             if (e.pageY<(C.height/2)) dbg_paused = true;     //DEBUG
272             else dbg_paused = false;                         //DEBUG
273         });                                                  //DEBUG
274 
275     };
276 }                                                    //DEBUG
